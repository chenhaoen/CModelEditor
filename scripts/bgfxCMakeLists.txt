cmake_minimum_required(VERSION 3.20)
project(bgfx)

file(GLOB BGFX_SOURCES 
    ${CMAKE_CURRENT_SOURCE_DIR}/include/bgfx/*.h
    ${CMAKE_CURRENT_SOURCE_DIR}/src/*.cpp
    ${CMAKE_CURRENT_SOURCE_DIR}/src/*.h
)

list(REMOVE_ITEM BGFX_SOURCES 
    ${CMAKE_CURRENT_SOURCE_DIR}/src/amalgamated.cpp
)

add_library(${PROJECT_NAME} STATIC ${BGFX_SOURCES})

target_include_directories(${PROJECT_NAME} PUBLIC 
    ${CMAKE_CURRENT_SOURCE_DIR}/include
    ${CMAKE_CURRENT_SOURCE_DIR}/3rdparty
    ${CMAKE_CURRENT_SOURCE_DIR}/3rdparty/dxsdk/include
    ${CMAKE_CURRENT_SOURCE_DIR}/3rdparty/khronos
)

target_link_libraries(${PROJECT_NAME} PRIVATE 
    bx
    bimg
)

add_compile_definitions(__STDC_FORMAT_MACROS)

set_target_properties(${PROJECT_NAME} PROPERTIES
    CXX_STANDARD 17
    CXX_STANDARD_REQUIRED ON
    CXX_EXTENSIONS OFF
)
